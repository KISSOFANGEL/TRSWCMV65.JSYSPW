function TRSTreeItem_loadChildren(_oTRSTree, _sParentId, _oCurrItem){

    //获取参数

    var nChannelId = 0, nSiteId = 0;

    if(_sParentId.subString(0, 1) == "S"){

        nSiteId = parseInt(_sParentId.subString(1));

    }else{

        nChannelId = parseInt(_sParentId);

    }

   

    //发出请求,获取频道列表的XML信息

    var oAction = new CTRSAction("../channel/channels_get.jsp");

    oAction.setParameter("ChannelId", nChannelId);

    oAction.setParameter("SiteId", nSiteId);   

    var sChannelsXML = oAction.doXMLHttpAction();

 

    //解析XML

    var arWCMObjs = WCMObjHelper.fromXMLString(sChannelsXML);

 

    //构造子栏目地址

    for(var i=0; i<arWCMObjs.length; i++){

        _oTRSTree.addItem(arWCMObjs[i].getProperty("ChnlName"), arWCMObjs[i].getProperty("ChannelId"), arWCMObjs[i].getProperty("ChnlDesc"), null, _sParentId);

    }  

}
