/**

 * 以对话框的方式提交页面请求

 *

 * @param _nWidth

 *            对话框宽度

 * @param _nHeight

 *            对话框高度

 * @param _oArgs

 *            传入给页面的参数,被打开的页面可以通过window.dialogArguments

 *

 * @return

 *          如果被打开的页面设置了window.dialogArguments,就可以将其作为返回值

 */

 

 function testDialogAction(){

        var nWidth = 400, nHeight = 300, oArgs;

        var oTRSAction = new CTRSAction("../test/action_page.html");

        oTRSAction.setParameter("ChannelId", 1);

        oTRSAction.setParameter("DocId", 1023);

        oTRSAction.doDialogAction(nWidth, nHeight, oArgs);

 }

 

 function testXMLHttpAction(){

        var oTRSAction = new CTRSAction("http://localhost:9000/wcm/index/index.jsp");

        oTRSAction.doXMLHttpAction();

 }

 

 

 

 function testDialogByObject(){

        var nWidth = 600, nHeight = 400, oArgs = ["", "", m_sLinkXML];

        CTRSAction_alert(oReturn);

        if(oReturn != null){

            CTRSAction_alert("type of:"+typeof(oReturn));

            CTRSAction_alert("0:"+oReturn[0]);        

            CTRSAction_alert("1:"+oReturn[1]);        

        }

 }
